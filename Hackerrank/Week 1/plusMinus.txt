void plusMinus(int arr_count, int* arr) {
    
    float zr,pr,nr,f;
    f=arr_count;
    int z=0,p=0,n=0,i;
    for(i=0;i<arr_count;i++)
    {
        if(arr[i]==0){
            z++;
        }
        else if(arr[i]>0){
            p++;
        }
        else{
            n++;
        }
    }

    zr=z/f;
    pr=p/f;
    nr=n/f;
    printf("%.6f\n",pr);
    printf("%.6f\n",nr);
    printf("%.6f",zr);
    
}
